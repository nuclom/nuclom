## Styling System Overview

Nuclom uses a modern CSS-in-JS approach with:

- **Tailwind CSS** for utility-first styling
- **CSS Variables** for dynamic theming
- **shadcn/ui** for component design system
- **next-themes** for theme management

## Theme Management

### Theme Provider

```typescript
// src/components/theme-provider.tsx
"use client";

import { ThemeProvider as NextThemesProvider } from "next-themes";
import { type ThemeProviderProps } from "next-themes/dist/types";

export function ThemeProvider({ children, ...props }: ThemeProviderProps) {
  return <NextThemesProvider {...props}>{children}</NextThemesProvider>;
}
```

### Theme Toggle

```typescript
// src/components/theme-toggle.tsx
"use client";

import { Moon, Sun } from "lucide-react";
import { useTheme } from "next-themes";
import { Button } from "@/components/ui/button";

export function ThemeToggle() {
  const { theme, setTheme } = useTheme();

  return (
    <Button
      variant="ghost"
      size="icon"
      onClick={() => setTheme(theme === "light" ? "dark" : "light")}
    >
      <Sun className="h-[1.2rem] w-[1.2rem] rotate-0 scale-100 transition-all dark:-rotate-90 dark:scale-0" />
      <Moon className="absolute h-[1.2rem] w-[1.2rem] rotate-90 scale-0 transition-all dark:rotate-0 dark:scale-100" />
      <span className="sr-only">Toggle theme</span>
    </Button>
  );
}
```

### App Layout Integration

```typescript
// src/app/layout.tsx

export default function RootLayout({

### Utility Classes

  "rounded-lg border p-4",
  variant === "primary" && "border-primary",
  variant === "secondary" && "border-secondary",
  size === "sm" && "p-2",
  size === "lg" && "p-6",
  className
)} />
```

### Variant-based Styling

```typescript
// Component with variants
interface ButtonProps {
  variant?: "default" | "destructive" | "outline" | "secondary" | "ghost" | "link";
  size?: "default" | "sm" | "lg" | "icon";
  className?: string;
}

const buttonVariants = {
  variant: {
    default: "bg-primary text-primary-foreground hover:bg-primary/90",
    destructive: "bg-destructive text-destructive-foreground hover:bg-destructive/90",
    outline: "border border-input bg-background hover:bg-accent hover:text-accent-foreground",
    secondary: "bg-secondary text-secondary-foreground hover:bg-secondary/80",
    ghost: "hover:bg-accent hover:text-accent-foreground",
    link: "text-primary underline-offset-4 hover:underline",
  },
  size: {
    default: "h-10 px-4 py-2",
    sm: "h-9 rounded-md px-3",
    lg: "h-11 rounded-md px-8",
    icon: "h-10 w-10",
  },
};

export function Button({ variant = "default", size = "default", className, ...props }: ButtonProps) {
  return (
    <button
      className={cn(
        "inline-flex items-center justify-center rounded-md text-sm font-medium transition-colors",
        buttonVariants.variant[variant],
        buttonVariants.size[size],
        className
      )}
      {...props}
    />
  );
}
```

### Dynamic Theme Variables

```typescript
// Dynamic theme management
export function useCustomTheme() {
  const setThemeColors = (colors: Record<string, string>) => {
    const root = document.documentElement;

    Object.entries(colors).forEach(([key, value]) => {
      root.style.setProperty(`--${key}`, value);
    });
  };

  const resetTheme = () => {
    const root = document.documentElement;
    const themeVars = [
      "primary",
      "primary-foreground",
      "secondary",
      "secondary-foreground",
      "accent",
      "accent-foreground",
    ];

    themeVars.forEach(varName => {
      root.style.removeProperty(`--${varName}`);
    });
  };

  return { setThemeColors, resetTheme };
}
```

## Best Practices

<Note>
**CSS Organization**
1. Use CSS layers for proper cascade control
2. Group related styles in components
3. Use semantic class names
4. Avoid deep nesting
5. Leverage CSS custom properties
</Note>

<Note>
**Theming Guidelines**
1. Use semantic color names (primary, secondary, etc.)
2. Provide fallback values for custom properties
3. Test themes in both light and dark modes
4. Consider accessibility when choosing colors
5. Use consistent spacing across themes
</Note>

<Tip>
**Performance Tips**
1. Minimize CSS bundle size with purging
2. Use CSS-in-JS for dynamic styles
3. Avoid inline styles when possible
4. Leverage browser caching for static styles
5. Use CSS variables for theme switching
</Tip>
